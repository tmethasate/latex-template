% note the addition of -optionfile=texcount.cfg in the command below
\newcommand{\wordcount}[1]{%
	\immediate\write18{texcount -incbib -nobib -sum #1 | grep 'Sum count:' | awk '{print $3}' > #1-words.sum}%
	\raggedleft\textit{\textbf{Total Words} (Excluding Footnotes and In-text Citations Counted as One): \textbf{\input{#1-words.sum}}}%
}

	\input{tcfootnotes}
	
	\wordcount{tcfootnotes.tex}
